70void waitForConnection(uint16_t streamID) {



71    struct timespec timeToWait;



72    struct timeval now;



73    double blockingTimeout;



74    ConnectionInfo *cliConn = getConnectionInfo(streamID);



75    



76    gettimeofday(&now,NULL);



77    if (cliConn->blockingTimeout < 0)



78    {



79        blockingTimeout = MW_TCP_DEFAULT_TIMEOUT;



80    }



81    else



82    {



83        blockingTimeout = cliConn->blockingTimeout;



84    }



85    



86    timeToWait.tv_sec = now.tv_sec + floor(blockingTimeout);



87    timeToWait.tv_nsec = (now.tv_usec + ((blockingTimeout - floor(blockingTimeout))*pow(10,6)))*1000UL;



88    



89    LOG(LOG_DEBUG,"waitForConnection: lock mutex\n");



90    pthread_mutex_lock(&cliConn->connect_mutex);



91    LOG(LOG_DEBUG,"calling pthread_cond_timedwait\n");



92    pthread_cond_timedwait(&cliConn->connect_cv, &cliConn->connect_mutex, &timeToWait);



93    LOG(LOG_DEBUG,"pthread_cond_timedwait finished. Unlock the mutex\n");



94    pthread_mutex_unlock(&cliConn->connect_mutex);



95}

97int checkDataAvailable(ConnectionInfo *connection, size_t size) {



98    int dataAvailable = 0;



99    struct timeval timeNow;



100    double blockingTimeout,startTime,currentTime,diffTime;



101    if (connection->blockingTimeout < 0)



102    {



103        connection->blockingTimeout = MW_TCP_DEFAULT_TIMEOUT;



104        blockingTimeout = MW_TCP_DEFAULT_TIMEOUT;



105    }



106    else



107    {



108        blockingTimeout = connection->blockingTimeout;



109    }



110    



111    ioctl(connection->sock, FIONREAD, &dataAvailable);



112    LOG(LOG_DEBUG,"checkDataAvailable: dataAvailable at recv buff = %d\n",dataAvailable);



113    /* Wait until data is available if socket is in blocking mode */



114    if ((dataAvailable < size) && (blockingTimeout > 0.0))



115    {



116        gettimeofday(&timeNow,NULL);



117        startTime = timeNow.tv_sec+(timeNow.tv_usec/1000000.0);



118        LOG(LOG_DEBUG, "checkDataAvailable: start while loop for waiting %f streamID = %d\n",blockingTimeout,connection->streamID);



119        do {



120            ioctl(connection->sock, FIONREAD, &dataAvailable);



121            /*LOG(LOG_DEBUG,"dataAvailable at recv buff = %d",dataAvailable);*/



122            if(dataAvailable >= size)



123                break;



124            usleep(1000);



125            gettimeofday(&timeNow,NULL);



126            currentTime = timeNow.tv_sec+(timeNow.tv_usec/1000000.0);



127            diffTime = currentTime - startTime;



128            /*LOG(LOG_DEBUG,"while loop running difftime = %f, dataAvailable = %d",diffTime,dataAvailable);*/



129        } while(diffTime < blockingTimeout);



130    }



131    return dataAvailable;



132}

134/* Function: initTCPConnection



135 * Initialize a client-server connection



136 * A return value of TCPSTREAM_ERROR indicates an error.



137 */



138static int16_t initTCPConnection(int connectionID,



139                                 ConnectionMode operMode,



140                                 SOCKET sock,



141                                 double blockingTimeout,



142                                 uint16_t portNumber,



143                                 char *hostName,



144                                 uint16_t localClientPort) {



145    int16_t retVal = TCPSTREAM_NO_ERROR;



146    ConnectionInfo *connection = &connectionInfoArray[connectionID];



147    



148    /* initialize the new ConnectionInfo */



149    connection->isInUse = 1;



150    connection->operMode = operMode;



151    if (1 == isinf(blockingTimeout))



152    {



153        connection->blockingTimeout = (double)INT_MAX/1000.0;



154    }



155    else



156    {



157        connection->blockingTimeout = blockingTimeout;



158    }



159    /* initialize to NULL early so that calls to



160     * freeConnection on error will succeed */



161    connection->serverData = NULL;



162    connection->SendBufferSize = -1;



163    connection->ReceiveBufferSize = -1;



164    



165    if (operMode == SERVER_MODE) {



166        /* initialize server data */



167        connection->serverData = (ServerData *)malloc(sizeof(ServerData));



168        if (connection->serverData == NULL) {



169            LOG(LOG_INFO, "initTCPConnection: serverData malloc failed\n");



170            freeConnection(connection);



171            retVal = TCP_ERR_NOCONN_SERVER;



172            return retVal;



173        }



174        connection->serverData->port = portNumber;



175        /* provided sock is the listening sock */



176        connection->serverData->listenSock = sock;



177        /* later call to serverAcceptConnection will set sock */



178        connection->sock = INVALID_SOCKET;



179        /*



180         *Save this info to reuse the connection.



181         * g1502287



182         */



183        connection->remotePort = portNumber;



184        sprintf(connection->remoteHost,"%s",hostName);



185    } else {



186        /* store the provided socket */



187        connection->sock = INVALID_SOCKET;



188        connection->clientSock = sock;



189        connection->remotePort = portNumber;



190        sprintf(connection->remoteHost,"%s",hostName);



191        connection->localClientPort = localClientPort;



192    }



193    LOG(LOG_INFO, "Connection id %d \n", connectionID);



194    



195    /* display the size of the socket receive buffer */



196    socklen_t optionLen = sizeof(int);



197    int optionValue;



198    getsockopt(sock, SOL_SOCKET, SO_SNDBUF, (char *) &optionValue, &optionLen);



199    LOG(LOG_INFO, "Connection id %d, SendBufSize: %d \n", connectionID, optionValue);



200    getsockopt(sock, SOL_SOCKET, SO_RCVBUF, (char *) &optionValue, &optionLen);



201    LOG(LOG_INFO, "Connection id %d, RecvBufSize: %d \n", connectionID, optionValue);



202    



203    LOG(LOG_INFO, "Connection id %d, blockingTimeout: %f \n", connectionID,



204        connection->blockingTimeout);



205    if (connection->operMode == SERVER_MODE)



206        LOG(LOG_INFO, "Connection id %d, type: server \n", connectionID);



207    else



208        LOG(LOG_INFO, "Connection id %d, type: client \n", connectionID);



209    



210    LOG(LOG_INFO, "Connection id %d, socket id %d \n", connectionID, (int) sock);



211    /* relevant to both clients and servers */



212    LOG(LOG_INFO, "Connection id %d, server port: %d \n", connectionID, portNumber);



213    return retVal;



214}

216/* Function: getConnectionInfo



217 * Retrieve a ConnectionInfo instance given its connectionID



218 * (as returned by initTCPConnection)



219 * NOTE: An invalid connectionID will lead to a NULL pointer being returned



220 */



221ConnectionInfo * getConnectionInfo(int connectionID) {



222    /* return NULL for invalid or uninitialized connectionIDs */



223    /*LOG(LOG_INFO,"getConnectionInfo form streamid = %d \n",connectionID);*/



224    ConnectionInfo *connection = NULL;



225    if ((connectionID >= 0) && (connectionID < MAX_NUM_CONNECTIONS)) {



226        if (connectionInfoArray[connectionID].isInUse)



227            connection = &connectionInfoArray[connectionID];



228        else



229            LOG(LOG_INFO, "getConnectionInfo: connection not in use.\n");



230    }



231    return connection;



232}

234/* Function: getConnectionIdx



235 * Returns a connectionID corresponding to a ConnectionInfo that is not



236 * already in use.



237 * A TCPSTREAM_ERROR will be returned if all available ConnectionInfo instances



238 * are already in use.



239 */



240static uint16_t getConnectionIdx(void) {



241    uint16_t connectionID;



242    int foundUnusedConnectionInfo = 0;



243    /* linear search for an unused ConnectionInfo */



244    for (connectionID = 0; connectionID < MAX_NUM_CONNECTIONS; connectionID++) {



245        if (!connectionInfoArray[connectionID].isInUse) {



246            foundUnusedConnectionInfo = 1;



247            break;



248        }



249    }



250    if (!foundUnusedConnectionInfo) {



251        /* all ConnectionInfo's are in use */



252        LOG(LOG_INFO, "getConnectionIdx: all %d available connections are in use\n", MAX_NUM_CONNECTIONS);



253        connectionID = TCPSTREAM_ERROR;



254    }



255    LOG(LOG_INFO, "getConnectionIdx: id = %d\n", connectionID);



256    return connectionID;



257}

259/* Function: freeConnection



260 * Frees memory associated with the ConnectionInfo referenced by connectionID.



261 * Marks the ConnectionInfo instance as no longer in use.



262 */



263static void freeConnection(ConnectionInfo *connection) {



264    /* mark the ConnectionInfo as not in use */



265    connection->isInUse = 0;



266    connection->blockingTimeout = -1;



267    connection->sock = -1;



268    connection->clientSock = -1;



269    /* free dynamic memory */



270    if (connection->operMode == SERVER_MODE) {



271        free(connection->serverData);



272        connection->serverData = NULL;



273    }



274}

277/* Function: checkSocketPending



278 *  Returns true, via the 'pending' arg, if data is pending on the comm line.



279 *  Returns false otherwise.



280 *  TCPSTREAM_NO_ERROR is returned on success, TCPSTREAM_ERROR on failure.



281 */



282static int checkSocketPending(



283                              const SOCKET sock,



284                              int *outPending,



285                              double blockingTimeout) {



286    struct pollfd ReadFds;



287    int pending;



288    int tout;



289    int retVal = TCPSTREAM_NO_ERROR;



290    



291    LOG(LOG_INFO, "checkSocketPending: blockingTimeout = %f\n", blockingTimeout);



292    if (1 == isinf(blockingTimeout)) {



293        tout = -1; /*infinite blocking - -1 */



294	} else if (blockingTimeout < 0) {



295		tout = MW_TCP_DEFAULT_TIMEOUT; /* non-blocking, return immediately */



296	}



297    else {



298        tout = 1000*blockingTimeout; /* poll-timeout is in milliseconds */



299    }



300    



301    memset(&ReadFds, 0, sizeof(ReadFds));



302    ReadFds.fd = sock;



303    ReadFds.events = POLLIN;



304    pending = poll(&ReadFds, 1, tout);



305    if (pending == SOCK_ERR)



306        retVal = TCPSTREAM_ERROR;



307    *outPending = (pending==1);



308    return (retVal);



309}

311/* Function: socketDataRecv



312 *  Attempts to gets the specified number of bytes from the specified socket.



313 *  The number of bytes read is returned via the 'recvStatus' parameter.



314 *  TCPSTREAM_NO_ERROR is returned on success, TCPSTREAM_ERROR is returned on



315 *  failure.



316 */



317static int16_t socketDataRecv(ConnectionInfo *connection, void *dst, size_t size, int8_t *recvStatus) {



318    



319    size_t nRead = 0;



320    int16_t retVal = TCPSTREAM_NO_ERROR;



321    int dataAvailable = 0;



322    



323    dataAvailable = checkDataAvailable(connection, size);



324    



325    if((size > 0) && (dataAvailable >= size)){



326        /*Read from recv buffer*/



327        errno = 0;



328        nRead = recv(connection->sock, dst, size, 0U);



329        /* dst data type depends on block mask selection. Comment out hardcoded uint8_t*/



330        LOG(LOG_INFO, "socketDataRecv: dataRead = %d, nRead = %d, errno = %d\n", *(uint8_t*)dst, nRead, errno);



331        if((nRead == 0) && (errno == 0)){



332            /*peer disconnected. Return error to CLOSE the socket*/



333            retVal = (connection->operMode ==  SERVER_MODE) ? TCP_ERR_NOCLIENT : TCP_ERR_NOSERVER;



334            return retVal;



335        }



336        if (nRead == SOCK_ERR) {



337            if(errno == ECONNRESET || errno == ENOTCONN)



338                retVal = (connection->operMode ==  SERVER_MODE) ? TCP_ERR_NOCLIENT : TCP_ERR_NOSERVER;



339        } else



340            retVal = TCPSTREAM_NO_ERROR;



341        



342        if (retVal==TCPSTREAM_NO_ERROR) {



343            if(nRead == size)



344                *recvStatus = RECEIVE_SUCCESS; /*Received requested amount of data. Set status as 1*/



345        }



346    } else {



347        /*Requested data not available.



348         * Use recv to check connection is active or not



349         */



350        LOG(LOG_DEBUG, "socketDataRecv: requested data not available, check connection using recv\n");



351        errno = 0;



352        /* Recv with size 0 returns zero though connection is good.  So minimum expected size is always 1.



353           Since, recv is called with MSG_PEEK, the recv data buffer is unchanged. */



354        if (size < 1)



355            size = 1;



356        nRead = recv(connection->sock, dst, size,  MSG_PEEK | MSG_DONTWAIT);



357        /* dst data type depends on block mask selection. Comment out hardcoded uint8_t*/



358        LOG(LOG_INFO, "socketDataRecv: dataRead = %d, nRead = %d, errno = %d \n",*(uint8_t*)dst, nRead, errno);



359        if((nRead == 0) && (errno == 0)){



360            /*peer disconnected. Return error to CLOSE the socket*/



361            retVal = (connection->operMode ==  SERVER_MODE) ? TCP_ERR_NOCLIENT : TCP_ERR_NOSERVER;



362        }



363        /*memset dst to zero to outupt 0*/



364        memset(dst,0,size);



365    }



366    



367    return retVal;



368}

370/* Function: socketDataSend



371 * Utility function to send data via the specified socket



372 */



373static int16_t socketDataSend(ConnectionInfo *connection, const void *src, const size_t size) {



374    int nSent;



375    int retVal = TCPSTREAM_NO_ERROR;



376    const void *sendSrc = src;



377    



378    errno = 0;



379#ifndef __APPLE__



380    nSent = send(connection->sock, (send_buffer_t)sendSrc, size, MSG_NOSIGNAL);



381#else



382    /* For iOS, MSG_NOSIGNAL flag is not defined. We need to set SO_NOSIGPIPE via setsockopt */



383    const int set = 1;



384    setsockopt(connection->sock, SOL_SOCKET, SO_NOSIGPIPE, &set, sizeof(int));



385    nSent = send(connection->sock, (send_buffer_t)sendSrc, size, 0);



386#endif



387    LOG(LOG_INFO, "socketDataSend: call to send returned: %d, errno: %d \n", nSent, errno);



388    if (nSent == SOCK_ERR) {



389        if (errno == ECONNRESET || errno == EPIPE || errno == ENOTCONN)



390            retVal = (connection->operMode == SERVER_MODE) ? TCP_ERR_NOCLIENT : TCP_ERR_NOSERVER;



391    }



392    return retVal;



393}

395/* Function: createListenSocket



396 * Opens the listening socket to be used for accepting an incoming connection.



397 */



398static SOCKET createListenSocket(uint16_t port, int16_t *errorNo) {



399    



400    struct sockaddr_in serverAddr;



401    int sockStatus;



402    socklen_t sFdAddSize = (socklen_t) sizeof(struct sockaddr_in);



403    SOCKET lFd;



404    int option;



405    



406    LOG(LOG_INFO, "createListenSocket: create a socket to listen for incoming connections\n");



407    memset((void *) &serverAddr,0,(size_t)sFdAddSize);



408    serverAddr.sin_family      = AF_INET;



409    serverAddr.sin_port        = htons((unsigned short int) port);



410    serverAddr.sin_addr.s_addr = htonl(INADDR_ANY);



411    



412    lFd = socket(AF_INET, SOCK_STREAM, 0);



413    



414    if (lFd == INVALID_SOCKET) {



415        *errorNo = TCP_ERR_NOCONN_SERVER;



416        LOG(LOG_INFO, "createListenSocket: call to socket() failed\n");



417    } else {



418        /*



419         * Listening socket should always use the SO_REUSEADDR option



420         * ("Unix Network Programming - Networking APIs:Sockets and XTI",



421         * Volume 1, 2nd edition, by W. Richard Stevens).



422         */



423        LOG(LOG_INFO, "createListenSocket: set socket option SO_REUSEADDR\n");



424        option = 1;



425        sockStatus =



426        setsockopt(lFd,SOL_SOCKET,SO_REUSEADDR,(char*)&option,sizeof(option));



427        if (sockStatus == SOCK_ERR) {



428            LOG(LOG_INFO, "createListenSocket: setsocketopt() call failed\n");



429            close(lFd);



430            lFd = INVALID_SOCKET;



431            *errorNo = TCP_ERR_NOCONN_SERVER;



432        }



433        



434        if (lFd != INVALID_SOCKET) {



435            /* Disable Nagle's Algorithm*/



436            option = 1;



437            sockStatus =



438            setsockopt(lFd,IPPROTO_TCP,TCP_NODELAY,(char*)&option,sizeof(option));



439            if (sockStatus == SOCK_ERR) {



440                LOG(LOG_INFO, "createListenSocket: setsocketopt() TCP_NODELAY call failed\n");



441                close(lFd);



442                *errorNo = TCP_ERR_NOCONN_SERVER;



443                lFd = INVALID_SOCKET;



444            }



445        }



446    }



447    



448    LOG(LOG_INFO, "createListenSocket: bind socket\n");



449    if (lFd != INVALID_SOCKET) {



450        sockStatus = bind(lFd, (struct sockaddr *) &serverAddr, sFdAddSize);



451        if (sockStatus == SOCK_ERR) {



452            LOG(LOG_INFO, "createListenSocket: bind() call failed: %d\n", errno);



453            close(lFd);



454            lFd = INVALID_SOCKET;



455            *errorNo = TCP_ERR_SOCBIND;



456        }



457    }



458    



459    LOG(LOG_INFO, "createListenSocket: listen for conncetions on socket\n");



460    if (lFd != INVALID_SOCKET) {



461        sockStatus = listen(lFd, 2); /*TO DO: check for backlog arg*/



462        if (sockStatus == SOCK_ERR) {



463            LOG(LOG_INFO, "createListenSocket: listen() call failed\n");



464            close(lFd);



465            lFd = INVALID_SOCKET;



466            *errorNo = TCP_ERR_NOCONN_SERVER;



467        }



468    }



469    



470    return lFd;



471}

473/* Function: serverAcceptConnection



474 *  Called when the target is not currently connected to the host, this



475 *  function attempts to open the connection.



476 *



477 *  In the case of sockets, this is a passive operation in that the host



478 *  initiates contact, the target simply listens for connection requests.



479 */



480static int serverAcceptConnection(ConnectionInfo *connection) {



481    struct sockaddr_in clientAddr;



482    socklen_t sFdAddSize = sizeof(struct sockaddr_in);



483    SOCKET cFd = INVALID_SOCKET;



484    int error = TCPSTREAM_NO_ERROR;



485    int pending;



486    



487    /* Check that the listening socket is still valid and open a new socket if not */



488    if (connection->serverData->listenSock == INVALID_SOCKET) {



489        int16_t error = 0;



490        connection->serverData->listenSock = createListenSocket(connection->serverData->port, &error);



491    }



492    



493    /* pass listenSock rather than sock */



494    LOG(LOG_INFO, "serverAcceptConnection: call checkSocketPending\n");



495    error = checkSocketPending(connection->serverData->listenSock,



496                               &pending,



497                               connection->blockingTimeout);



498    LOG(LOG_INFO, "serverAcceptConnection: pending = %d\n", pending);



499    



500    if ((pending > 0) && (error==TCPSTREAM_NO_ERROR)) {



501        LOG(LOG_INFO, "serverAcceptConnection: wait to accept incoming connection\n");



502        cFd = accept(connection->serverData->listenSock,



503                     (struct sockaddr *)&clientAddr,



504                     &sFdAddSize);



505        



506        if (cFd == INVALID_SOCKET) {



507            LOG(LOG_INFO, "serverAcceptConnection: call to accept() for socket failed\n");



508            error = TCPSTREAM_ERROR;



509        }



510        else {



511            /* set sock */



512            /*First client to get connected to the server.



513             *assign this value within the thread. connection->sock = cFd;



514             *TO DO



515             *Stop further connections



516             */



517        }



518        LOG(LOG_INFO, "serverAcceptConnection: descriptor for accepted socket, cFd = %d\n", cFd);



519        



520        if (error == TCPSTREAM_ERROR) {



521            close(connection->serverData->listenSock);



522            connection->serverData->listenSock = INVALID_SOCKET;



523        }



524    }



525    return cFd;



526}

528/* Function: hostnameLookup



529 * Lookup target network name.



530 */



531static unsigned long hostnameLookup(char *hostName) {



532    



533    struct hostent *hp = NULL;



534    struct in_addr *iaddr = NULL;



535    unsigned long addr = INADDR_NONE;



536    



537    /* Default to localhost if hostname not specified.*/



538    if (hostName == NULL) {



539        static char localhost[] = "localhost";



540        hostName = localhost;



541    }



542    



543    /*



544     * See if the address is an IPV4 dot separated address:



545     */



546    addr = inet_addr(hostName);



547    



548    if (addr == INADDR_NONE) {



549        /* Since the address is not an IPV4 dot separated address,



550         * do a name lookup to get this:



551         */



552        hp = gethostbyname(hostName);



553        if (hp == NULL) {



554            LOG(LOG_INFO, "hostnameLookup: gethostbyname() call failed\n");



555            addr = INADDR_NONE;



556        } else {



557            iaddr = (struct in_addr *) hp->h_addr;



558            addr = iaddr->s_addr;



559        }



560    }



561    return(addr);



562}

565/* Function: connectionLookup



566 * Lookup for existing connections.



567 */



568static int8_t connectionLookup(char *hostName, uint16_t port) {



569    int8_t i, streamIndex = -1;



570    if (hostName == NULL) return streamIndex;



571    



572    for(i = 0; i < MAX_NUM_CONNECTIONS; i++) {



573        if(connectionInfoArray[i].isInUse == 1) {



574            if((strcmp(connectionInfoArray[i].remoteHost, hostName) == 0) && (connectionInfoArray[i].remotePort == port)){



575                streamIndex = i;



576                break;



577            }



578        }



579    }



580    return streamIndex;



581}

583/* Function: createClientSocket



584 * Create a socket for client connection and bind if required



585 */



586static SOCKET createClientSocket(char *hostName, uint16_t portNum, uint16_t clientPortToBind, int16_t *errorNo) {



587    



588    struct sockaddr_in remoteAddr,localAddr;



589    unsigned long addr = INADDR_NONE;



590    SOCKET cSock = INVALID_SOCKET;



591    addr = hostnameLookup(hostName);



592    



593    if(addr != INADDR_NONE) {



594        remoteAddr.sin_addr.s_addr = addr;



595        remoteAddr.sin_family = AF_INET; /*hp->h_addrtype;*/



596        remoteAddr.sin_port   = htons((unsigned short) portNum);



597        



598        LOG(LOG_INFO, "createClientSocket: create client socket\n");



599        cSock = socket(PF_INET,SOCK_STREAM,0);



600        if (cSock == INVALID_SOCKET) {



601            *errorNo = TCP_ERR_NOCONN_CLIENT;



602            LOG(LOG_INFO, "createClientSocket: socket() call failed\n");



603            return cSock;



604        } else {



605            /*bind to a local port if configured */



606            if(clientPortToBind > 0){



607                localAddr.sin_addr.s_addr = INADDR_ANY;



608                localAddr.sin_family = AF_INET;



609                localAddr.sin_port   = htons((unsigned short) clientPortToBind);



610                



611                if(bind(cSock, (struct sockaddr *) &localAddr, sizeof(localAddr)) < 0){



612                    LOG(LOG_INFO, "createClientSocket: local port binding failed\n");



613                    close(cSock);



614                    *errorNo = TCP_ERR_SOCBIND;



615                    return INVALID_SOCKET;



616                }



617            }



618        }



619    } else {



620        /* Error in resolving host address */



621        *errorNo = TCP_ERR_NOHOST;



622    }



623    return cSock;

703void *clientConn_handler(void *input) {



704    uint16_t streamId = *(uint16_t *)input;



705    ConnectionInfo *cliConn = getConnectionInfo(streamId);



706    if (cliConn == NULL) {



707        LOG(LOG_INFO, "clientConn_handler: cliConn is NULL, streamId: %d\n", streamId);



708        return NULL;



709    }



710    struct sockaddr_in remoteAddr;



711    LOG(LOG_INFO, "clientConn_handler: start %d\n", streamId);



712    



713    remoteAddr.sin_addr.s_addr = hostnameLookup(cliConn->remoteHost);



714    remoteAddr.sin_family = AF_INET; /*hp->h_addrtype;*/



715    remoteAddr.sin_port   = htons((unsigned short) cliConn->remotePort);



716    if (remoteAddr.sin_addr.s_addr == INADDR_NONE)



717        return NULL;



718    while(1) {



719        if(connect(cliConn->clientSock, (struct sockaddr *)&remoteAddr, sizeof(remoteAddr)) == 0) {



720            LOG(LOG_INFO, "clientConn_handler: %d connect success\n", streamId);



721            



722            /* set the user specified timeout for socket */



723            setTimeoutForSocket(cliConn->clientSock, cliConn->blockingTimeout);



724            



725            /* Change Buffer size */



726            setReceiveBufferSize(cliConn->clientSock,cliConn->ReceiveBufferSize);



727            setSendBufferSize(cliConn->clientSock,cliConn->SendBufferSize);



728            



729            pthread_mutex_lock(&cliConn->connect_mutex);



730            cliConn->sock = cliConn->clientSock;



731            



732            /* signal the wait for connection timed waiting to wakeup the step function.*/



733            pthread_cond_signal(&cliConn->connect_cv);



734            



735            /* wait here till the connection is lost.*/



736            LOG(LOG_INFO, " clientConn_handler: %d, pthread cond wait\n", streamId);



737            pthread_cond_wait(&cliConn->connect_cv, &cliConn->connect_mutex);



738            LOG(LOG_INFO, "clientConn_handler: %d, wake from sleep\n", streamId);



739            pthread_mutex_unlock(&cliConn->connect_mutex);



740        }



741#ifdef __APPLE__



742        /* failure to connect invalidates the socket descriptor on BSD derived systems.



743         So, we need to create a new client socket before calling connect*/



744        else {



745            int errorNo = 1;



746            close(cliConn->clientSock);



747            cliConn->clientSock = createClientSocket(cliConn->remoteHost, cliConn->remotePort,cliConn->localClientPort, errorNo);



748        }



749#endif



750        sleep(1);



751        LOG(LOG_INFO, "clientConn_handler: %d, while loop running\n", streamId);



752    }



753}

755void *serverConn_handler(void *input) {



756    uint16_t streamId = *(uint16_t *)input;



757    ConnectionInfo *serConn = getConnectionInfo(streamId);



758    if (serConn == NULL) {



759        LOG(LOG_INFO, "serverConn_handler: serConn is NULL, streamId: %d\n", streamId);



760        return NULL;



761    }



762    SOCKET cFd = 0;



763    



764    LOG(LOG_INFO,"serverConn_handler: start \n");

767        if(cFd > 0){



768            /* connection success. save it to struct */



769            LOG(LOG_INFO,"serverConn_handler: connect success\n");



770            setTimeoutForSocket(cFd, serConn->blockingTimeout);



771            /* Change Buffer size */



772            setReceiveBufferSize(cFd,serConn->ReceiveBufferSize);



773            setSendBufferSize(cFd,serConn->SendBufferSize);



774            



775            pthread_mutex_lock(&serConn->connect_mutex);

781        }



782        sleep(1);



783        LOG(LOG_INFO,"serverConn_handler: while loop running \n");



784    }



785    LOG(LOG_INFO,"serverConn_handler: end \n");



786}

799void TCPStreamSetup(uint16_t portNumber, uint16_t localClientPort, uint16_t *connStream, uint16_t server, double blockingTimeout, int16_t *errorNo, char *hostName) {

808    LOG(LOG_INFO, "TCPStreamSetup: stream setup hostName = %s, portNumber = %d, connectionMode = %s \n", hostName, portNumber, (server == 0) ? "Client" : "Server");



809

819

828



829        /* Update time out */



830        connData = getConnectionInfo(*connStream);

833        if (1 == isinf(connData->blockingTimeout))

840

899void TCPStreamStepSend(void *dataIn, size_t dataInSize, uint16_t connStream, uint16_t isSer, int16_t *errorNo) {



900    LOG(LOG_INFO, "TCPStreamStepSend: connStream = %d, dataInSize = %d\n", connStream, dataInSize);



901    ConnectionInfo *connData = NULL;



902    int pending = 0;



903    if(connStream >= 0){



904        connData = getConnectionInfo(connStream);



905        LOG(LOG_INFO,"TCPStreamStepSend: check connection and send data\n");



906        if (connData == NULL) {



907            *errorNo = (isSer == 1) ? TCP_ERR_NOCONN_SERVER : TCP_ERR_NOCONN_CLIENT;



908            return;



909        }



910        if(connData->sock <= 0){



911            /*No connection. Threads are running to connect to client/server.



912             * Block the step function if required.



913             */



914            if(connData->operMode == SERVER_MODE) {



915                /*block on listen socket*/



916                LOG(LOG_INFO, "TCPStreamStepSend: invalid socket, block on listen socket\n");



917                checkSocketPending(connData->serverData->listenSock, &pending, connData->blockingTimeout);



918            }else {



919                /*block on client socket */



920                LOG(LOG_DEBUG, "TCPStreamStepSend: connStream %d wait for a successful connection\n", connStream);



921                waitForConnection(connStream);



922            }



923        }



924        if(connData->sock > 0) {



925            /*valid connection. Send data*/



926            *errorNo = socketDataSend(connData, dataIn, dataInSize);



927            



928            if((*errorNo == TCP_ERR_NOCLIENT) || (*errorNo == TCP_ERR_NOSERVER)){



929                LOG(LOG_INFO, "TCPStreamStepSend: error in connection, close the socket\n");



930                close(connData->sock);



931                LOG(LOG_INFO, "TCPStreamStepSend: lock the mutex and signal the thread\n");



932                pthread_mutex_lock(&connData->connect_mutex);



933                connData->sock = INVALID_SOCKET;



934                



935                /*recreate socket if in client mode */



936                if(connData->operMode == CLIENT_MODE){



937                    LOG(LOG_INFO, "TCPStreamStepSend: re-create new socket for client connection\n");



938                    connData->clientSock = createClientSocket(connData->remoteHost, connData->remotePort,connData->localClientPort,errorNo);



939                }



940                LOG(LOG_INFO, "TCPStreamStepSend: signal the thread to reconnect\n");



941                pthread_cond_signal(&connData->connect_cv);



942                pthread_mutex_unlock(&connData->connect_mutex);



943                LOG(LOG_INFO, "TCPStreamStepSend: Unlock the mutex\n");



944            }



945        } else



946            *errorNo = (connData->operMode == SERVER_MODE) ? TCP_ERR_NOCLIENT : TCP_ERR_NOSERVER;



947    }



948    if((*errorNo != TCP_ERR_NOCLIENT) && (*errorNo != TCP_ERR_NOSERVER)) {



949        *errorNo = TCP_ERR_CONNSUCCESS;



950        LOG(LOG_INFO, "TCPStreamStepSend: Sent successfully\n");



951    }



952



953}
